achievement_magnanimous = {
	possible = {
		c:BRZ ?= THIS
		has_dlc_feature = rp1_content
	}
	happened = {
		custom_tooltip = {
			text = achievement_magnanimous_tt
			has_variable = brz_enduring_empire
		}
		has_law = law_type:law_monarchy
	} 
}

achievement_desteapta_te_romane = {
	possible = {
		country_has_primary_culture = cu:romanian
	}
	happened = {
		custom_tooltip = {
			text = achievement_desteapta_te_romane_tt
			c:ROM ?= this
		}
		custom_tooltip = {
			text = achievement_desteapta_te_romane_2_tt
			any_state = {
				filter = {
					state_region = {
						is_homeland = cu:romanian
					}
				}
				is_split_state = no
				owner = ROOT
				count = all
			}
		}
	}
}

achievement_piratini_not_piratiny = {
	possible = {
		OR = {
			c:PNI ?= THIS
			c:PRA ?= THIS
		}
	}
	happened = {
		custom_tooltip = {
			text = achievement_piratini_not_piratiny_tt
			OR = {
				NOT = {
					exists = c:BRZ
				}
				c:BRZ ?= {
					this.gdp < root.gdp
				}
			}
		}
	}
}

achievement_stonks = {
	happened = {
		hidden_trigger = {
			any_company = {
				prosperity >= 100
				count >= 3
			}
		}
	}
}

achievement_banana_republic = {
	possible = {
		OR = {
			country_has_primary_culture = cu:yankee
			country_has_primary_culture = cu:dixie
		}
	}
	happened = {
		hidden_trigger = {
			any_company = {
				type = company_type:company_united_fruit
				prosperity >= 200
			}
			country_has_building_group_levels = {
				type = bg_banana_plantations
				value  >= 75
			}
		}
	}
}

			}
		}
	}
