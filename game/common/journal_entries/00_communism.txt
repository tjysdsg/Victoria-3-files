je_communism_1 = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	is_shown_when_inactive = {
		NOT = { has_variable = communism_1_done }
		OR = {
			has_technology_researched = socialism
			has_technology_researched = anarchism
		}
	}

	possible = {
		any_interest_group = {
			OR = {
				is_being_bolstered = yes
				is_in_government = yes
			}
			leader = {
				OR = {
					has_ideology = ideology_communist
					has_ideology = ideology_vanguardist
					has_ideology = ideology_anarchist
				}
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_variable = {
				name = communism_je_1_var
				value = 0
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0
			5 = communism.7
		}
		effect = {
			if = {
				limit = {
					any_interest_group = {
						is_in_government = yes
						leader = {
							OR = {
								has_ideology = ideology_communist
								has_ideology = ideology_vanguardist
								has_ideology = ideology_anarchist
							}
						}
					}
				}
				change_variable = {
					name = communism_je_1_var
					add = 1
				}
			}
		}
	}

	complete = {
		has_technology_researched = anarchism
		has_technology_researched = human_rights
		has_technology_researched = socialism
		scope:journal_entry = {
			is_goal_complete = yes
		}
		any_interest_group = {
			is_in_government = yes
			ig_clout > 0.15
			leader = {
				OR = {
					has_ideology = ideology_communist
					has_ideology = ideology_vanguardist
					has_ideology = ideology_anarchist
				}
			}
		}
	}

	on_complete = {
		set_variable = communism_1_done
		custom_tooltip = custom_trigger_event_unspecific
		hidden_effect = {
			trigger_event = { id = communism.1 popup = yes }
		}
	}

	current_value = {
		value = root.var:communism_je_1_var
	}

	goal_add_value = {
		add = 3
	}

	progressbar = yes

}
