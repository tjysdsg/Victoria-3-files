usa_purchase_alaska = {
	is_shown = {
		exists = c:USA
		exists = c:RUS
		c:USA = ROOT
		is_subject = no
		NOT = {
			has_variable = alaska_purchase_var
		}
		c:RUS = {
			owns_entire_state_region = STATE_ALASKA
			is_subject = no
			is_player = no
		}
	}

	possible = {
		owns_entire_state_region = STATE_WASHINGTON
		in_default = no
		taking_loans = no
		is_at_war = no
		root.gdp > 60000000
		c:RUS = {
			relations:root >= relations_threshold:cordial
			is_at_war = no
			OR = {
				in_default = yes
				taking_loans = yes
			}
		}
	}

	when_taken = {
		set_variable = {
			name = alaska_purchase_var
			value = yes
		}
		trigger_event = {
			id = alaska.4
			days = 0
			popup = yes
		}
	}

	ai_chance = {
		base = 0	
		modifier = {
			trigger = { gold_reserves > 3000000 }
			add = 10
		}		
		modifier = {
			trigger = { 
				gold_reserves > 0
				weekly_net_fixed_income > 100000
			}
			add = 10
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}
	}
}
