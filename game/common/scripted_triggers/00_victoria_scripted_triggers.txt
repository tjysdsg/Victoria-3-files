law_is_valid_for_ig_petition = {
	save_temporary_scope_as = petition_law_trig
	prev = {
		law_stance = {
			law = scope:petition_law_trig.type
			value > neutral
		}
	}
	NOT = {
		ROOT = {
			has_law = scope:petition_law_trig.type
		}
	}  
	ROOT = { 
		can_have_political_movement = scope:petition_law_trig.type
	} 			
	can_be_enacted = yes
}
