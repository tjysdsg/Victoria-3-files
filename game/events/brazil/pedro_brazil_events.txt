	type = country_event
	placement = ROOT

	title = pedro.1.t

	flavor = pedro.1.f

		video = "votp_monarch_holding_court"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	dlc = dlc008

	trigger = {
		 # triggered by je_pedro_brazil
		 NOT = { has_variable = pedro_startup_var }
	}

	immediate = {
		set_variable = pedro_startup_var
		random_scope_character = {
			limit = {
				has_variable = is_pedro
			}
			save_scope_as = pedro_scope
		}
		ig:ig_landowners = { # old regency
			save_scope_as = landowners_scope
		}
		ig:ig_armed_forces = { # old regency
			save_scope_as = armed_forces_scope
		}
		ig:ig_intelligentsia = { # new regency
			save_scope_as = intelligentsia_scope
			leader = {
				save_scope_as = feijo_regent
			}
		}
	}

	option = {
		name = pedro.1.a
		default_option = yes
		custom_tooltip = {
			text = pedro_regency_party_tt
		}
		add_modifier = {
			name = brz_regency

		}
	}
}

	dlc = dlc008

		# triggered by je_pedro_brazil
		OR = {
			AND = {
				NOT = { has_variable = pedro_lo_marginal_var }

					is_marginal = yes
				}
			}
			AND = {
				NOT = { has_variable = pedro_af_marginal_var }

			}

			}
		}
		else = {
			set_variable = pedro_af_marginal_var

				save_scope_as = marginal_ig_scope
			}
		}
		ruler = {

		}
	}

	option = {
		name = pedro.5.a

pedro.8 = {
	type = country_event
	placement = ROOT

	title = pedro.8.t
	desc = pedro.8.d
	flavor = pedro.8.f

	event_image = {
		video = "southamerica_aristocrats"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	dlc = dlc008

	trigger = {
		# triggered by je_pedro_brazil
	}

	immediate = {
		ig:ig_armed_forces = {
			save_scope_as = armed_forces_ig
		}
		ig:ig_landowners = {
			save_scope_as = landowners_ig
		}
		ruler = {
			save_scope_as = pedro_scope
		}
		ROOT = {
			save_scope_as = brazil_scope
		}
		if = {
			limit = {
				NOT = { has_law = law_type:law_landed_voting }
			}
			activate_law = law_type:law_landed_voting
		}
		activate_law = law_type:law_presidential_republic
	}

	option = {
		name = pedro.8.a
		default_option = yes
		hidden_effect = {
			ig:ig_landowners = {
				remove_ideology = ideology_stratocratic
				add_ideology = ideology_oligarchic
			}
		}
		if = {
			limit = {
				ig:ig_landowners = {
					is_in_government = no
					is_marginal = no
				}
			}
			ig:ig_landowners = {
				add_ruling_interest_group = yes
			}
		}
		if = {
			limit = {
				ig:ig_armed_forces = {
					is_in_government = no
					is_marginal = no
				}
			}
			ig:ig_armed_forces = {
				add_ruling_interest_group = yes
			}
		}
		if = {
			limit = {
				ig:ig_armed_forces = {
					is_marginal = no
				}
			}
			ig:ig_armed_forces = {
				leader = {
					set_character_as_ruler = yes
				}
			}
		}
		else = {
			ig:ig_landowners = {
				leader = {
					set_character_as_ruler = yes
				}
			}
		}
		add_radicals = {
			value = large_radicals
		}
		every_country = {
			limit = {
				NOT = {
					THIS = ROOT
				}
				has_interest_marker_in_region = region_brazil
			}
			post_notification = brz_brazil_coup
		}
		set_variable = {
			name = abolishing_monarchy_var
			value = 0
		}
		add_journal_entry = { type = je_abolish_monarchy }
	}
}

pedro.12 = {
	type = country_event
	placement = ROOT

	title = pedro.12.t
	desc = pedro.12.d
	flavor = pedro.12.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	duration = 3

	dlc = dlc008

	trigger = {
		NOT = { has_variable = pedro_heir_death }
		year > 1849
		any_scope_character = {
			is_heir = yes
		}
	}

	immediate = {
		set_variable = pedro_heir_death
		ruler = {
			save_scope_as = pedro_scope
		}
		heir = {
			save_scope_as = heir_scope
		}
	}

	option = { # No women allowed
		name = pedro.12.a
		default_option = yes
		create_character = {
			template = brz_isabel_template
			heir = yes
		}
		# appease the landowners slightly
		ig:ig_landowners = {
			add_modifier = {
				name = princess_isabel
				days = long_modifier_time
				multiplier = -1
			}
		}
		remove_pedropoint_effect = yes
	}
	option = { # Prepare isabel for the throne
		name = pedro.12.b
		default_option = yes
		remove_pedropoint_effect = yes
		create_character = {
			template = brz_isabel_template
			heir = yes
			on_created = {
				save_scope_as = isabel_scope
				set_variable = isabel_var
				add_modifier = {
					name = isabel_heir_popularity
				}
				set_character_immortal = yes
			}
		}
		# pedro's popularity is reduced
		scope:pedro_scope = {
			add_modifier = {
				name = isabel_heir_popularity
			}
		}
		# annoy the landowners
		ig:ig_landowners = {
			add_modifier = {
				name = princess_isabel
				days = long_modifier_time
				multiplier = 3
				is_decaying = yes
			}
		}
		# add isabel JE
		set_variable = {
			name = isabel_points
			value = 0
		}
		add_journal_entry = { type = je_isabel target = scope:isabel_scope }
		custom_tooltip = {
			text = isabel_je_tt
		}
	}
}
