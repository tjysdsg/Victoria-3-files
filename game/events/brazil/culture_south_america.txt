		}

			}

		}
		capital = {
			ordered_scope_pop = {
				limit = {
					culture = scope:old_south_american_culture
				}
				order_by = strata
				change_pop_culture = {
					target = scope:new_national_culture
					value = 0.75
				}
			}
		}
	}
}

culture_south_america.2 = {
	type = country_event
	placement = scope:south_american_culture_state

	title = culture_south_america.2.t
	desc = culture_south_america.2.d
	flavor = culture_south_america.2.f

		video = "unspecific_ruler_speaking_to_people"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	}

	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				state_region = {
					is_homeland = scope:old_national_culture
				}
				any_neighbouring_state = {
					state_region = {
						is_homeland = scope:new_national_culture
					}
					OR = {
						is_capital = yes
						custom_tooltip = {
							text = new_culture_has_25_percent_tt
							culture_percent_state = {
								target = scope:new_national_culture
								value >= 0.25
							}
						}
					}
				}
				OR = {
					any_scope_building = {
						OR = {
							is_building_type = building_naval_base
							is_building_type = building_barracks
						}
						level >= 15
					}
					any_scope_building = {
						is_building_type = building_university
						level >= 5
					}
					any_scope_building = {
						is_building_type = building_government_administration
						level >= 5
					}
					any_scope_building = {
						is_building_type = building_urban_center
						level >= 5
					}
				}
			}
			random_neighbouring_state = {
				limit = {
					state_region = {
						is_homeland = scope:new_national_culture
					}
					OR = {
						is_capital = yes
						custom_tooltip = {
							text = new_culture_has_25_percent_tt
							culture_percent_state = {
								target = scope:new_national_culture
								value >= 0.25
							}
						}
					}
					any_neighbouring_state = {
						state_region = {
							is_homeland = scope:old_national_culture
						}
					}
				}
				save_scope_as = south_american_culture_state
			}
		}
	}

	option = {
		name = culture_south_america.2.a
		default_option = yes
		scope:south_american_culture_state = {
			if = {
				limit = {
					any_neighbouring_state = {
						owner = root
						state_region = {
							is_homeland = scope:old_national_culture
							NOT = {
								is_homeland = scope:new_national_culture
							}
						}
					}
				}
				every_neighbouring_state = {
					limit = {
						owner = root
						state_region = {
							is_homeland = scope:old_national_culture
							NOT = {
								is_homeland = scope:new_national_culture
							}
						}
						is_incorporated = yes
						OR = {
							any_scope_building = {
								OR = {
									is_building_type = building_naval_base
									is_building_type = building_barracks
								}
								level >= 15
							}
							any_scope_building = {
								is_building_type = building_university
								level >= 5
							}
							any_scope_building = {
								is_building_type = building_government_administration
								level >= 5
							}
