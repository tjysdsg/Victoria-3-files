namespace = tunguska

tunguska.1 = {
	type = country_event
	placement = scope:tunguska_state
	
	title = tunguska.1.t
	desc = tunguska.1.desc
	flavor = tunguska.1.f
	
	event_image = {
		video = "unspecific_fire"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"
	
	duration = 3
	
	trigger = {
		game_date >= 1908.1.1
		NOT = { has_global_variable = tunguska_var }
		any_scope_state = {
			state_region = s:STATE_UPPER_YENISEYSK
		}
	}

	immediate = {
		set_global_variable = tunguska_var
		random_scope_state = {
			limit = {
				state_region = s:STATE_UPPER_YENISEYSK
			}
			save_scope_as = tunguska_state
		}
	}

	option = { # interesting, but irrelevant
		name = tunguska.1.a
		default_option = yes
		s:STATE_UPPER_YENISEYSK = {
			add_devastation = 20
		}
	}

	option = { # study the crater
		name = tunguska.1.b
		highlighted_option = yes
		trigger = {
			ig:ig_intelligentsia = {
				is_in_government = yes
			}
		}
		add_modifier = {
			name = tunguska_research
			months = short_modifier_time
		}
		s:STATE_UPPER_YENISEYSK = {
			add_devastation = 20
		}
	}
	option = { # it is an omen
		name = tunguska.1.c
		highlighted_option = yes
		trigger = {
			ig:ig_devout = {
				is_in_government = yes
			}
		}
		ig:ig_devout = {
			add_modifier = {
				name = tunguska_omen
				months = short_modifier_time
			}
		}
		s:STATE_UPPER_YENISEYSK = {
			add_devastation = 20
		}
	}
	option = { # can we mine it?
		name = tunguska.1.d
		highlighted_option = yes
		trigger = {
			ig:ig_industrialists = {
				is_in_government = yes
			}
		}
		scope:tunguska_state = {
			add_modifier = {
				name = tunguska_mining
				months = short_modifier_time
			}
		}
		s:STATE_UPPER_YENISEYSK = {
			add_devastation = 20
		}
	}
}
