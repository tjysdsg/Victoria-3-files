# Pebrine Outbreak Start
silkworm_diseases.1 = {
	type = country_event
	placement = scope:pebrine_state
	event_image = {
		video = "asia_dead_cattle_poor_harvest"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	cooldown = { months = normal_modifier_time }

	title = silkworm_diseases.1.t
	desc = silkworm_diseases.1.d
	flavor = silkworm_diseases.1.f

	duration = 3

	dlc = dlc004

	trigger = {
		has_dlc_feature = voice_of_the_people_content
		any_scope_state = {
			is_incorporated = yes
			any_scope_building = { 
	            is_building_type = building_silk_plantation
	            level >= 5
	        }
		}
		OR = {
			can_research = pasteurization
			is_researching_technology = pasteurization
		}
		OR = {	
			country_is_in_europe = yes
			country_is_in_north_america = yes
		}
		NOT = {
			has_variable = pebrine_outbreak_event
		}	
	}

	immediate = {
		set_variable = pebrine_outbreak_event
		random_scope_state = {
			limit = {
				any_scope_building = { 
		            is_building_type = building_silk_plantation
		            level >= 5
		        }
			}
			save_scope_as = pebrine_state
			random_scope_building = {
				limit = {
					is_building_type = building_silk_plantation
		            level >= 5
				}
				save_scope_as = pebrine_building
			}
		}
	}

	option = {
		name = silkworm_diseases.1.a
		default_option = yes
		add_journal_entry = {
			type = je_silkworm_diseases
		}
		scope:pebrine_state = {
			add_modifier = {
				name = modifier_pebrine_state_modifier
				months = long_modifier_time
			}
		}
	}
}

	desc = silkworm_diseases.3.d

	flavor = silkworm_diseases.3.f

	duration = 3

	dlc = dlc004

	trigger = {
		#Triggered by the JE
	}

	immediate = {
		set_variable = pebrine_outbreak_event
		if = {
			limit = {
				any_neighbouring_state = {
				is_incorporated = yes
					owner = {
						any_scope_state = {
							is_incorporated = yes
							has_modifier = modifier_pebrine_state_modifier
						}
					}
				}
			}
			random_neighbouring_state = {
				limit = {
					is_incorporated = yes
					owner = {
						any_scope_state = {
							is_incorporated = yes
							has_modifier = modifier_pebrine_state_modifier
						}
					}			
				}
				owner = {
					save_scope_as = pebrine_origin
				}
			}
		}
		else_if = {
			limit = {
				any_country = {
					NOT = { THIS = ROOT }
					market = ROOT.market
					any_scope_state = {
						has_modifier = modifier_pebrine_state_modifier
					}
				}
			}
			random_country = {
				limit = {
					NOT = { THIS = ROOT }
					market = ROOT.market
					any_scope_state = {
						has_modifier = modifier_pebrine_state_modifier
					}
				}
				save_scope_as = pebrine_origin
			}
		}
		else = {
			random_country = {
				limit = {
					NOT = { THIS = ROOT }
					has_journal_entry = je_silkworm_diseases
				}
				save_scope_as = pebrine_origin
			}
		}
		random_scope_state = {
			limit = {
				any_scope_building = { 
		            is_building_type = building_silk_plantation
		            level >= 2
		        }
			}
			save_scope_as = pebrine_state
			random_scope_building = {
				limit = {
					is_building_type = building_silk_plantation
		            level >= 2
				}
				save_scope_as = pebrine_building
			}
		}
	}

	option = {
		name = silkworm_diseases.3.a
		default_option = yes
		add_journal_entry = {
			type = je_silkworm_diseases
		}
		scope:pebrine_state = {
			add_modifier = {
				name = modifier_pebrine_state_modifier
				months = long_modifier_time
			}
		}
	}
	option = {
		name = silkworm_diseases.3.b
		add_journal_entry = {
			type = je_silkworm_diseases
		}
		change_relations = {
			country = scope:pebrine_origin
			value = -10
		}
		scope:pebrine_state = {
			add_modifier = {
				name = modifier_pebrine_state_modifier
				months = long_modifier_time
			}
		}
	}

	desc = silkworm_diseases.4.d

	trigger = {
		any_scope_state = {
			has_modifier = modifier_pebrine_state_modifier
		}

			is_marginal = no
		}
		OR = {
			has_law = law_type:law_serfdom
			has_law = law_type:law_tenant_farmers
		}
	}

	immediate = {
		random_scope_state = {
			limit = {

			}
			save_scope_as = pebrine_state
			random_scope_building = {
				limit = {
					is_building_type = building_silk_plantation
				}
				save_scope_as = pebrine_building
			}
		}
		ig:ig_landowners = {
			save_scope_as = landowners_scope
		}
	}

	option = {
		name = silkworm_diseases.4.a
		default_option = yes
		add_modifier = {
			name = modifier_landowner_bailout_funds
			months = short_modifier_time
			multiplier = money_amount_multiplier_small
		}
		add_modifier = {
			name = modifier_landowner_bailout_funds_country
			months = short_modifier_time
		}
		scope:pebrine_state = {
			add_loyalists_in_state = {
				value = small_radicals

			}
		}
	}

		name = silkworm_diseases.4.b
		scope:pebrine_state = {
			add_radicals_in_state = {
				value = small_radicals

			}
